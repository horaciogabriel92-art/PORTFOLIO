Why Fisher-Yates?

The Fisher-Yates shuffle produces a uniformly random permutation. Each element has exactly 1/n probability of ending up in any position. This is crucial for a fair card draw.

fisher-yates.js 
function fisherYatesShuffle(array) {
  // Create a copy to avoid mutating original
  const arr = [...array];
  
  // Iterate from last to first element
  for (let i = arr.length - 1; i > 0; i--) {
    // Pick random index from 0 to i
    const j = Math.floor(Math.random() * (i + 1));
    
    // Swap elements at i and j
    [arr[i], arr[j]] = [arr[j], arr[i]];
  }
  
  return arr;
}

const deck = ['The Fool', 'The Magician', ...];
const shuffled = fisherYatesShuffle(deck);
const selection = [shuffled[0], shuffled[1]];

CSS 3D Transforms

The card flip uses transform-style: preserve-3d and backface-visibility: hidden. The container has perspective to create depth perception.

card-flip.css 
/* Container establishes 3D context */
.card-wrapper {
  perspective: 1000px;
}

/* Inner card enables 3D transforms */
.card-inner {
  transform-style: preserve-3d;
  transition: transform 0.8s cubic-bezier(0.4, 0, 0.2, 1);
}

/* Flip the card */
.card-inner.is-flipped {
  transform: rotateY(180deg);
}

/* Front face: hidden when flipped */
.card-front, .card-back {
  backface-visibility: hidden;
  position: absolute;
  inset: 0;
}

/* Back face starts rotated */
.card-back {
  transform: rotateY(180deg);
}

State Machine Pattern

The system uses a state machine to manage UI transitions: IDLE → SHUFFLING → REVEALING → COMPLETE. This prevents race conditions.

state-machine.js 
const TarotSystem = {
  states: {
    IDLE: 'idle',
    SHUFFLING: 'shuffling',
    REVEALING: 'revealing',
    COMPLETE: 'complete'
  },
  
  current: 'idle',
  
  transition(to) {
    const valid = {
      idle: ['shuffling'],
      shuffling: ['revealing'],
      revealing: ['complete'],
      complete: ['idle']
    };
    
    if (valid[this.current].includes(to)) {
      this.current = to;
      this.execute(to);
    }
  }
};

function checkStrength(pass) {
  let score = 0;
  const tests = [
    { regex: /.{8,}/, name: 'length' },
    { regex: /[A-Z]/, name: 'uppercase' },
    { regex: /[0-9]/, name: 'number' },
    { regex: /[^A-Za-z0-9]/, name: 'special' }
  ];
  
  tests.forEach(test => {
    if (test.regex.test(pass)) score++;
  });
  
  return score; // 0-4 scale
}

Canvas Particle System 

class Particle {
  constructor(x, y) {
    this.x = x;
    this.y = y;
    this.vx = (Math.random() - 0.5) * 2;
    this.vy = (Math.random() - 0.5) * 2;
  }
  
  update() {
    this.x += this.vx;
    this.y += this.vy;
  }
}

Debounce Function

const debounce = (fn, delay) => {
    let timer;
    return (...args) => {
        clearTimeout(timer);
        timer = setTimeout(() => fn.apply(this, args), delay);
    };
};
